Parameters

adLink : "pass in the redirectionLink"

originallink : "pass in the correct URL"

mobilead: "pass in the redirectionLink for mobile Users"

desktopad : "pass in the redirectionLink for desktop Users"

local: "pass in the value "clear" to clear the localStorage "
